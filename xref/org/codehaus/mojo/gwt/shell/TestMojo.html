387     @Override
388     public void doExecute()
389         throws MojoExecutionException, MojoFailureException
390     {
391         if ( skip || skipTests || skipExec )
392         {
393             return;
394         }
395         new TestTemplate( getProject(), includes, excludes, new TestTemplate.CallBack()
396         {
397             public void doWithTest( File sourceDir, String test )
398                 throws MojoExecutionException
399             {
400                 forkToRunTest( test );
401             }
402         } );
403 
404         if ( failures > 0 )
405         {
406             if ( testFailureIgnore )
407             {
408                 getLog().error( "There are test failures.\n\nPlease refer to " + reportsDirectory
409                                     + " for the individual test results." );
410             }
411             else
412             {
413                 throw new MojoExecutionException( "There was test failures." );
414             }
415         }
416     }

418     /**
419      * @param classpath the test execution classpath
420      * @param jvm the JVM process command
421      * @param test the test to run
422      * @throws MojoExecutionException some error occured
423      */
424     private void forkToRunTest( String test )
425         throws MojoExecutionException
426     {
427         test = test.substring( 0, test.length() - 5 );
428         test = StringUtils.replace( test, File.separator, "." );
429         try
430         {
431             File outFile = new File(out);
432             if (outFile.isAbsolute())
433             {
434                 outFile.mkdirs();
435             }
436             else
437             {
438                 new File( getProject().getBasedir(), out ).mkdirs();
439             }
440             try
441             {
442                 JavaCommand cmd = createJavaCommand()
443                     .setMainClass( MavenTestRunner.class.getName() );
444                 if ( gwtSdkFirstInClasspath )
445                 {
446                     cmd.addToClasspath( getGwtUserJar() )
447                        .addToClasspath( getGwtDevJar() );
448                 }
449                 cmd.addToClasspath( getClasspath( Artifact.SCOPE_TEST ) );
450                 if ( !gwtSdkFirstInClasspath )
451                 {
452                     cmd.addToClasspath( getGwtUserJar() )
453                        .addToClasspath( getGwtDevJar() );
454                 }
455 
456                 addCompileSourceArtifacts( cmd );
457 
458                 cmd.arg( test );
459                 cmd.systemProperty( "surefire.reports", reportsDirectory.getAbsolutePath() );
460                 cmd.systemProperty( "gwt.args", getGwtArgs() );
461 
462                 cmd.execute();
463             }
464             catch ( ForkedProcessExecutionException e )
465             {
466                 failures++;
467             }
468         }
469         catch ( Exception e )
470         {
471             throw new MojoExecutionException( "Failed to run GWT tests", e );
472         }
473     }

475     protected String getGwtArgs()
476     {
477         StringBuilder sb = new StringBuilder();
478         sb.append( "-war " ).append( quote( out ) );
479         sb.append( " -logLevel " ).append( quote( getLogLevel() ) );
480         sb.append( ( webMode || productionMode ) ? " -nodevMode" : " -devMode" );
481         sb.append( checkAssertions ? " -checkAssertions" : " -nocheckAssertions" );
482         sb.append( clusterFunctions ? " -XclusterFunctions" : " -XnoclusterFunctions" );
483         sb.append( disableCastChecking ? " -XnocheckCasts" : " -XcheckCasts" );
484         sb.append( disableClassMetadata ? " -XnoclassMetadata" : " -XclassMetadata" );
485         sb.append( disableRunAsync ? " -XnocodeSplitting" : " -XcodeSplitting" );
486         sb.append( draftCompile ? " -draftCompile" : " -nodraftCompile" );
487         sb.append( inlineLiteralParameters ? " -XinlineLiteralParameters" : " -XnoinlineLiteralParameters" );
488         sb.append( optimizeDataflow ? " -XoptimizeDataflow" : " -XnooptimizeDataflow" );
489         sb.append( ordinalizeEnums ? " -XordinalizeEnums" : " -XnoordinalizeEnums" );
490         sb.append( removeDuplicateFunctions ? " -XremoveDuplicateFunctions" : " -XnoremoveDuplicateFunctions" );
491         sb.append( showUi ? " -showUi" : " -noshowUi" );
492         sb.append( " -sourceLevel " ).append( quote( sourceLevel ) );
493         sb.append( " -testBeginTimeout " ).append( testBeginTimeout );
494         sb.append( " -testMethodTimeout ").append( testMethodTimeout );
495         sb.append( " -Xtries " ).append( tries );
496         sb.append( incremental ? " -incremental" : " -noincremental" );
497 
498         if ( optimizationLevel >= 0 )
499         {
500             sb.append( " -optimize " ).append( optimizationLevel );
501         }
502         if ( precompile != null && !precompile.trim().isEmpty() )
503         {
504             sb.append( " -precompile " ).append( quote( precompile ) );
505         }
506         if ( logDir != null )
507         {
508             sb.append( " -logdir " ).append( quote( logDir.getAbsolutePath() ) );
509         }
510         if ( workDir != null )
511         {
512             sb.append( " -workDir " ).append( quote( workDir.getAbsolutePath() ) );
513         }
514 
515         if ( namespace != null && !namespace.trim().isEmpty() )
516         {
517             sb.append( " -Xnamespace " ).append( quote( namespace ) );
518         }
519         if ( jsInteropMode != null && jsInteropMode.length() > 0 && !jsInteropMode.equals( "NONE" ) )
520         {
521             sb.append( " -XjsInteropMode " ).append( quote( jsInteropMode ) );
522         }
523 
524         if ( mode.equalsIgnoreCase( "manual" ) )
525         {
526             sb.append( " -runStyle Manual:1" );
527         }
528         else if ( mode.equalsIgnoreCase( "htmlunit" ) )
529         {
530             sb.append( " -runStyle ").append( quote( "HtmlUnit:" + htmlunit ) );
531         }
532         else if ( mode.equalsIgnoreCase( "selenium" ) )
533         {
534             sb.append( " -runStyle ").append( quote( "Selenium:" + selenium ) );
535         }
536         else if ( !mode.trim().isEmpty() )
537         {
538             sb.append( " -runStyle ").append( quote( mode ) );
539         }
540         if ( userAgents != null && !userAgents.trim().isEmpty() )
541         {
542             sb.append( " -userAgents " ).append( quote( userAgents ) );
543         }
544         if ( batch != null && !batch.trim().isEmpty() )
545         {
546             sb.append( " -batch " ).append( quote( batch ) );
547         }
548         // TODO Is addArgumentDeploy(cmd) also needed to get readable test stacktraces with an alternative deploy dir?
549 
550         return sb.toString();
551     }

553     private String quote(String arg) {
554         return StringUtils.quoteAndEscape( arg, '"', new char[] { '"', ' ', '\t', '\r', '\n' } );
555     }

557     @Override

564     /**
565      * @param clazz class to check for classpath resolution
566      * @return The classpath element this class was loaded from
567      */
568     private File getClassPathElementFor( Class<?> clazz )
569     {
570         String classFile = clazz.getName().replace( '.', '/' ) + ".class";
571         ClassLoader cl = Thread.currentThread().getContextClassLoader();
572         if ( cl == null )
573         {
574             cl = getClass().getClassLoader();
575         }
576         URL url = cl.getResource( classFile );
577         getLog().debug( "getClassPathElementFor " + clazz.getName() + " file " + url.toString() );
578         String path = url.toString();
579     
580         if ( path.startsWith( "jar:" ) )
581         {
582             path = path.substring( 4, path.indexOf( "!" ) );
583         }
584         else
585         {
586             path = path.substring( 0, path.length() - classFile.length() );
587         }
588         if ( path.startsWith( "file:" ) )
589         {
590             path = path.substring( 5 );
591             // windauze hack with maven 3 we get those !
592             path = path.replace( "%20", " " );
593         }
594         File file = new File( path );
595         getLog().debug( "getClassPathElementFor " + clazz.getName() + " file " + file.getPath() );
596         return file;
597     }
